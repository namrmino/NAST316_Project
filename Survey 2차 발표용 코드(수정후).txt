/*C1번 점수화 과정 [해성]*/
/*점수화 (산술)*/
DATA SURVEY;
SET SURVEY;
MEAN1 = (C1_1+C1_2+C1_3+C1_4)/4; 
RUN;
/*점수화 (가중)*/

quit;
